# deployment.yaml — Kubernetes 部署配置
apiVersion: apps/v1
kind: Deployment
metadata:
  name: data-pipeline-api
  namespace: data-pipeline
  labels:
    app: data-pipeline
    version: v2.3.1
    team: platform
spec:
  replicas: 3
  selector:
    matchLabels:
      app: data-pipeline
  template:
    spec:
      containers:
        - name: api
          image: registry.internal/data-pipeline:v2.3.1
          ports:
            - containerPort: 8000
          env:
            - name: JWT_SECRET
              valueFrom:
                secretKeyRef:
                  name: api-secrets
                  key: jwt-secret
            - name: DB_CONNECTION_STRING
              valueFrom:
                secretKeyRef:
                  name: db-credentials
                  key: mongo-uri
            - name: REDIS_URL
              value: "redis://redis-cluster-01.internal:6379/0"
            - name: LOG_LEVEL
              value: "INFO"
          resources:
            requests:
              memory: "256Mi"
              cpu: "250m"
            limits:
              memory: "512Mi"
              cpu: "500m"
          livenessProbe:
            httpGet:
              path: /health
              port: 8000
            initialDelaySeconds: 10
            periodSeconds: 30

---
# .github/workflows/deploy.yml
name: Deploy Data Pipeline
on:
  push:
    branches: [main]
    tags: ['v*']
jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Build & Push
        run: |
          docker build -t registry.internal/data-pipeline:${{ github.sha }} .
          docker push registry.internal/data-pipeline:${{ github.sha }}
      - name: Rollout
        run: |
          kubectl set image deployment/data-pipeline-api api=registry.internal/data-pipeline:${{ github.sha }} -n data-pipeline
          kubectl rollout status deployment/data-pipeline-api -n data-pipeline --timeout=300s
